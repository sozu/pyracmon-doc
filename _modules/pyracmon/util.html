[docs]def key_to_index(values: Dict[Union[str, int], Any], ordered_keys: List[Union[str, int]]) -> Dict[int, Any]:
    """
    Replace keys of a `dict` with its index in ordered list.

    :param values: A dictionary.
    :param ordered_keys: Ordered keys.
    :returns: New dictionary where keys are replaced.
    """
    def index(k):
        if isinstance(k, int):
            return k
        else:
            return ordered_keys.index(k)

    return {index(k):v for k, v in values.items()}
